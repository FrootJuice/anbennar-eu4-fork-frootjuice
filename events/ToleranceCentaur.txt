namespace = centaur_tolerance_events

# Event organizer for the centaur tolerance events - should be a huge improvement to performance - having 1 overall check for having the minority instead of one for EACH new event that gets added
country_event = {
	id = centaur_tolerance_events.0
	title = race_setup_events.1.d
	desc = race_setup_events.1.d
	picture = BORDER_TENSION_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		fast_owns_num_pops_of_race = { race = centaur amount = 1 }
		NOT = { has_country_modifier = centaur_administration }
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 10
			has_country_modifier = racial_focus_centaur
		}
	}

	option = {
		random_list = {
			1 = { }
			# ensure there's an empty option first in case none of the others are available
			# and also because it seems this one might be picked rarely even if the others are available...

			#generic demand autonomy
			10 = {
				trigger = {
					any_owned_province = {
						NOT = { local_autonomy = 50 }
						is_capital = no
						has_centaur_minority_trigger = yes
					}
				}
				country_event = { id = centaur_tolerance_events.1 }
			}
			#generic tolerance increase
			10 = {
				trigger = {
					OR = {
						stability = 1
						in_golden_age = yes
						ruler_has_personality = tolerant_personality
						ruler_has_personality = kind_hearted_personality
						ruler_has_personality = benevolent_personality
						has_idea_group = humanist_ideas
						
						any_ally = {
							has_country_modifier = centaur_administration
						}
					}
				}
				country_event = { id = centaur_tolerance_events.2 }
			}
			#generic tolerance decrease
			10 = {
				trigger = {
					OR = {
						NOT = { stability = 0 }
						is_bankrupt = yes
						inflation = 25
						AND = {
							is_at_war = yes
							NOT = { war_score = 20 }
						}
						war_exhaustion = 10
						ruler_has_personality = cruel_personality
						ruler_has_personality = malevolent_personality
						ruler_has_personality = conqueror_personality
						
						any_rival_country = {
							has_country_modifier = centaur_administration
						}
					}
				}
				country_event = { id = centaur_tolerance_events.3 }
			}
			#centaur minister
			10 = {
				modifier = {
					factor = 1.5
					medium_tolerance_centaur_race_trigger = yes
				}
				modifier = {
					factor = 2
					high_tolerance_centaur_race_trigger = yes
				}
				country_event = {
					id = centaur_tolerance_events.4
				}
			}
		}
	}
}

#Demand Representation
country_event = {
	id = centaur_tolerance_events.1
	title = centaur_tolerance_events.1.t
	desc = centaur_tolerance_events.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	goto = racial_pop_province_target
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_centaur_minority_trigger = yes
					is_capital = no
				}
				save_event_target_as = racial_pop_province_target
			}
			random_owned_province = {
				limit = {
					NOT = { local_autonomy = 50 }
					is_capital = no
					has_centaur_minority_trigger = yes
				}
				save_event_target_as = racial_pop_province_target
			}
			random_owned_province = {
				limit = {
					NOT = { local_autonomy = 50 }
					is_capital = no
					has_centaur_minority_trigger = yes
					development = 10
				}
				save_event_target_as = racial_pop_province_target
			}
		}
	}
	
	option = {		#Granted!
		name = centaur_tolerance_events.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_increase_tolerance_centaur = yes
			}
		}
		
		add_adm_power = -40
		
		medium_increase_of_centaur_tolerance_effect = yes
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 25
		}
	}
	
	option = {		#Make some concessions
		name = centaur_tolerance_events.1.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_maintain_tolerance_centaur = yes
			}
			modifier = {
				factor = 0.5
				average_autonomy = 10
			}
		}
		
		add_adm_power = -20
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 10
		}

	}
	
	option = {		#No. Haven't we given them enough?!
		name = centaur_tolerance_events.1.c
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_decrease_tolerance_centaur = yes
			}
		}
		
		medium_decrease_of_centaur_tolerance_effect = yes
		
	}
	
	option = {		#Special: Make empty promises
		name = centaur_tolerance_events.1.e
		trigger = { ruler_has_personality = charismatic_negotiator_personality }
		highlight = yes
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0.25
				wants_to_decrease_tolerance_centaur = yes
			}
			modifier = {
				factor = 1.5
				average_autonomy = 10
			}
		}
		
		add_dip_power = -10
		
		small_increase_of_centaur_tolerance_effect = yes
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 1
		}
		
	}
}

	}
	
}

# centaur wayfarer (centaur minister)
country_event = {
	id = centaur_tolerance_events.4
	title = centaur_tolerance_events.4.t
	desc = centaur_tolerance_events.4.d
	picture = ADVISOR_eventPicture
	
	is_triggered_only = yes

	# If you have a centaur culture province, that's the one the advisor comes from
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					any_owned_province = {
						culture_group = centaur
					}
				}
				random_owned_province = {
					limit = { culture_group = centaur }
					save_event_target_as = centaur_advisor_culture_province
				}
				set_country_flag = centaur_advisor_culture_set
			}
		}
	}

	# Recruit the centaur
	option = {
		name = centaur_tolerance_events.4.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				wants_to_maintain_tolerance_centaur = yes
			}
			modifier = {
				factor = 0.1
				wants_to_decrease_tolerance_centaur = yes
			}
		}
		small_increase_of_centaur_tolerance_effect = yes
		if = {
			limit = { has_country_flag = centaur_advisor_culture_set }
			if = {
				limit = { monthly_income = 50 }
				random_list = {
					1 = {
						define_advisor = {
							type = trader
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 3
							
							discount = yes
						}
					}
					1 = {
						define_advisor = {
							type = grand_captain
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 3
							
							discount = yes
						}
					}
				}
			}
			else_if = {
				limit = { monthly_income = 25 }
				random_list = {
					1 = {
						define_advisor = {
							type = trader
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 2							
							discount = yes
						}
					}
					1 = {
						define_advisor = {
							type = grand_captain
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 2							
							discount = yes
						}
					}
				}
			}
			else = {
				random_list = {
					1 = {
						define_advisor = {
							type = trader
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 1
							discount = yes
						}
					}
					1 = {
						define_advisor = {
							type = grand_captain
							culture = event_target:centaur_advisor_culture_province
							religion = event_target:centaur_advisor_culture_province
							skill = 1							
							discount = yes
						}
					}
				}
			}
		}
	}

	# A true exemplar of their race
	option = {
		name = centaur_tolerance_events.4.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				wants_to_increase_tolerance_centaur = yes
			}
		}
		medium_increase_of_centaur_tolerance_effect = yes
	}

	# Go away silly horse person
	option = {
		name = centaur_tolerance_events.4.c
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				wants_to_decrease_tolerance_centaur = yes
			}
		}
		small_decrease_of_centaur_tolerance_effect = yes
	}

	after = {
		clr_country_flag = centaur_advisor_culture_set
	}
}
